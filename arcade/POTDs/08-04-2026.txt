#include <iostream>
#include <string>
#include <algorithm>

using namespace std;

bool isSubsequence(const string &S, const string &T) {
    int j = 0;
    for (int i = 0; i < S.size() && j < T.size(); i++) {
        if (S[i] == T[j]) j++;
    }
    return j == T.size();
}

int find_smallest_day(string S, string T) {
    string curr = S;
    int day = 1;

    while (true) {
        if (isSubsequence(curr, T)) return day;

        string rev = curr;
        reverse(rev.begin(), rev.end());
        curr = rev + S;
        if (curr.size() > 4000) {
            curr = curr.substr(0, 2000) + curr.substr(curr.size() - 2000);
        }
        day++;
        if (day > 2000) break;
    }
    return -1;
}
