#include <iostream>
#include <string>
using namespace std;

int user_logic(int n, string s) {
    string tmp = "";

    for (int i = 0; i < n; i++) {
        tmp.push_back(s[i]);

        if (tmp.size() >= 4) {
            int size = tmp.size();
            if (tmp[size - 4] == 'k' && tmp[size - 3] == 'i' && tmp[size - 2] == 'l' && tmp[size - 1] == 'l') {
                tmp.pop_back();
                tmp.pop_back();
                tmp.pop_back();
                tmp.pop_back();
            }
        }
    }
    return tmp.size();
}
