#include <iostream>
#include <string>
using namespace std;

int user_logic(int n, string s) {
    int count = 0;

    for (int i = 0; i <= n - 4; i++) {
        if (s[i] == 'k' && s[i + 1] == 'i' && s[i + 2] == 'l' && s[i + 3] == 'l') {
            
            count++;
        }
    }
    return n - count;
}
