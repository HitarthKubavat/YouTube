#include <iostream>
#include <vector>
#include <string>
#include <set>
#include <map>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    if (!(cin >> N)) return 0;

    set<int> unique_drivers;
    map<string, int> restaurant_counts;
    int max_popularity_score = 0;

    for (int i = 0; i < N; ++i) {
        int driver_id;
        string restaurant_name;
        int base_rating;

        cin >> driver_id >> restaurant_name >> base_rating;

        unique_drivers.insert(driver_id);

        restaurant_counts[restaurant_name]++;

        int current_popularity = (int)restaurant_name.length() * base_rating;
        
        if (current_popularity > max_popularity_score) {
            max_popularity_score = current_popularity;
        }
    }

    string most_frequent_restaurant = "";
    int max_entries = 0;
    for (auto const& [name, count] : restaurant_counts) {
        if (count > max_entries) {
            max_entries = count;
            most_frequent_restaurant = name;
        }
    }

    cout << unique_drivers.size() << endl;
    cout << most_frequent_restaurant << endl;
    cout << max_popularity_score << endl;

    return 0;
}
