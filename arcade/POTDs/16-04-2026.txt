#include <iostream>
#include <vector>
using namespace std;

const int MOD = 1000000007;

vector<vector<int>> adj;
long long result = 1;

int dfs(int node, int parent, vector<int>& roles) {
    int sum = roles[node];

    for (int nei : adj[node]) {
        if (nei == parent) continue;

        int sub = dfs(nei, node, roles);

        if (sub > 0) {
            result = (result * sub) % MOD;
        }

        sum += sub;
    }

    return sum;
}

int calculate_valid_ways(int N, vector<int>& roles, vector<pair<int, int>>& connections) {
    adj.assign(N, vector<int>());
    result = 1;

    for (auto &edge : connections) {
        int u = edge.first;
        int v = edge.second;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    int totalArchitects = 0;
    for (int x : roles) totalArchitects += x;

    if (totalArchitects == 1) return 1;

    dfs(0, -1, roles);

    return result % MOD;
}

int main() {
    int N;
    cin >> N;

    vector<int> roles(N);
    for (int i = 0; i < N; ++i) {
        cin >> roles[i];
    }

    vector<pair<int, int>> connections(N-1);
    for (int i = 0; i < N - 1; ++i) {
        cin >> connections[i].first >> connections[i].second;
    }

    int result = calculate_valid_ways(N, roles, connections);
    cout << result << endl;

    return 0;
}
