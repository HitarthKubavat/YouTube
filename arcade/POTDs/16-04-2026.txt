#include <iostream>
#include <vector>
using namespace std;

const int MOD = 1000000007;

vector<vector<int>> adj;
int totalArchitects = 0;
long long result = 1;

int dfs(int node, int parent, vector<int>& roles) {
    int count = roles[node];

    for (int neighbor : adj[node]) {
        if (neighbor == parent) continue;

        int sub = dfs(neighbor, node, roles);

        if (sub == 1) {
            result = (result * 1) % MOD;
        }

        count += sub;
    }

    return count;
}

int calculate_valid_ways(int N, vector<int>& roles, vector<pair<int, int>>& connections) {
    
    adj.assign(N, vector<int>());

    for (auto &edge : connections) {
        adj[edge.first].push_back(edge.second);
        adj[edge.second].push_back(edge.first);
    }

    totalArchitects = 0;
    for (int x : roles) {
        if (x == 1) totalArchitects++;
    }

    if (totalArchitects == 1) return 1;

    result = 1;

    dfs(0, -1, roles);

    return result % MOD;
}
