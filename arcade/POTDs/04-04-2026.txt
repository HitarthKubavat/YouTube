#include <iostream>
#include <algorithm>
#include <vector>
using namespace std;

int minimum_sum(int N) {

    vector<int> numbers;
    while (N > 0) {
        numbers.push_back(N % 10);
        N /= 10;
    }


    sort(numbers.begin(), numbers.end());

    int minsum = 1e9;


    do {
        int no1 = numbers[0] * 10 + numbers[1];
        int no2 = numbers[2] * 10 + numbers[3];

        int sum = no1 * no2;

        if (sum < minsum) {
            minsum = sum;
        }
    } while (next_permutation(numbers.begin(), numbers.end()));

    return minsum;
}
