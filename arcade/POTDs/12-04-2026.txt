#include <iostream>
#include <vector>

using namespace std;

int calculateMaxHappiness(int N, int B, const vector<int>& entryFees, const vector<int>& happiness) {
    
    vector<int> p1(B + 1, 0);
    vector<int> p2(B + 1, 0);

    for (int i = 0; i < N; i++) {
        vector<int> curr = p1;

        for (int j = entryFees[i]; j <= B; j++) {
            curr[j] = max(curr[j], p2[j - entryFees[i]] + happiness[i]);
        }

        p2 = p1;
        p1 = curr;
    }

    return p1[B];
}
