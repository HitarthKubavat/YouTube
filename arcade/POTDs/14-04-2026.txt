#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

vector<int> processApplications(int N, int T, vector<vector<int>>& applications) {
    vector<int> result;

    for (int i = 0; i < N; i++) {
        vector<int>& scores = applications[i];

        sort(scores.begin(), scores.end(), greater<int>());

        int sum = 0;
        int count = 0;
        bool possible = false;

        for (int j = 0; j < scores.size(); j++) {
            sum += scores[j];
            count++;

            if (sum >= T) {
                possible = true;
                break;
            }
        }

        if (possible) {
            result.push_back(count);
        } else {
            result.push_back(-1);
        }
    }

    return result;
}
