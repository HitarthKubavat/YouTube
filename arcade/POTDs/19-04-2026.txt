#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

const int MAX_BITS = 30;

struct LinearBasis {
    int basis[MAX_BITS];
    int pos[MAX_BITS];

    LinearBasis() {
        for (int i = 0; i < MAX_BITS; i++) {
            basis[i] = 0;
            pos[i] = 0;
        }
    }

    void insert(int val, int p, const LinearBasis& prev) {
        *this = prev;
        for (int i = MAX_BITS - 1; i >= 0; i--) {
            if ((val >> i) & 1) {
                if (!basis[i]) {
                    basis[i] = val;
                    pos[i] = p;
                    return;
                }
                if (p > pos[i]) {
                    swap(val, basis[i]);
                    swap(p, pos[i]);
                }
                val ^= basis[i];
            }
        }
    }

    int query(int x, int L) {
        int res = x;
        for (int i = MAX_BITS - 1; i >= 0; i--) {
            if (pos[i] >= L) {
                // If flipping the bit reduces the value, XOR it
                if ((res ^ basis[i]) < res) {
                    res ^= basis[i];
                }
            }
        }
        return res;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, Q;
    if (!(cin >> N >> Q)) return 0;

    vector<LinearBasis> prefixes(N + 1);
    for (int i = 1; i <= N; i++) {
        int val;
        cin >> val;
        prefixes[i].insert(val, i, prefixes[i - 1]);
    }

    while (Q--) {
        int l, r, x;
        cin >> l >> r >> x;
        cout << prefixes[r].query(x, l) << "\n";
    }

    return 0;
}
